using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace pract5amain
{
    public partial class AutoPostBack : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            // Runs on every request. Use this to set defaults ONLY once:
            if (!IsPostBack)
            {
                Label5.Text = DropDownList1.SelectedValue; // show default course once
            }
        }

        // TextBox AutoPostBack event
        protected void TextButton1_TextChanged(object sender, EventArgs e)
        {
            Label1.Text = TextBox1.Text;
        }

        // DropDownList AutoPostBack event
        protected void DropDownList1_SelectedIndexChanged(object sender, EventArgs e)
        {
            Label5.Text = DropDownList1.SelectedValue;
        }

        // RadioButtonList AutoPostBack event
        protected void RadioButtonList1_SelectedIndexChanged(object sender, EventArgs e)
        {
            Label6.Text = RadioButtonList1.SelectedValue;
        }

        // Submit
        protected void Button1_Click(object sender, EventArgs e)
        {
            Label7.Text = "Submitted successfully!!!";
        }

        // Clear
        protected void Button2_Clear(object sender, EventArgs e)
        {
            TextBox1.Text = string.Empty;
            DropDownList1.SelectedIndex = 0;
            CheckBox1.Checked = CheckBox2.Checked =
            CheckBox3.Checked = CheckBox4.Checked = CheckBox5.Checked = false;
            cbMarathi.Checked = cbEnglish.Checked = false;
            RadioButtonList1.ClearSelection();

            Label1.Text = Label5.Text = Label6.Text = Label7.Text = string.Empty;
        }
    }
}

using System;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace pract5b
{
    public partial class form1 : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            if (!IsPostBack)
            {
                Calendar1.SelectedDate = DateTime.Today;
            }
        }
        protected void Button2_Click(object sender, EventArgs e)
        {
            Calendar1.Caption = "Shubham Dawal";
            Calendar1.FirstDayOfWeek = FirstDayOfWeek.Sunday;
            Calendar1.NextPrevFormat = NextPrevFormat.FullMonth;
            Calendar1.TitleFormat = TitleFormat.Month;
            Label2.Text = "Today's Date: " + Calendar1.SelectedDate.ToShortDateString();
            Label3.Text = "Diwali Vacation Start: 23-10-2025";
            DateTime diwali = new DateTime(2025, 10, 23);
            int daysToDiwali = (diwali - Calendar1.SelectedDate).Days;
            if (daysToDiwali >= 0)
                Label4.Text = "Days Remaining For Diwali Vacation: " + daysToDiwali;
            else
                Label4.Text = "Diwali Vacation already started/ended";
            DateTime newYear = new DateTime(2025, 12, 31);
            int daysToNewYear = (newYear - Calendar1.SelectedDate).Days;
            if (daysToNewYear >= 0)
                Label5.Text = "Days Remaining for New Year: " + daysToNewYear;
            else
                Label5.Text = "Happy New Year!";
            if (Calendar1.SelectedDate == new DateTime(2025, 10, 23))
                Label3.Text = "<b>Diwali Festival Start</b>";
            if (Calendar1.SelectedDate == new DateTime(2025, 10, 28))
                Label3.Text = "<b>Diwali Festival Ends</b>";
        }
        protected void Calender1_DayRender(object sender, DayRenderEventArgs e)
        {
            if (e.Day.Date.Day == 5 && e.Day.Date.Month == 9)
            {
                e.Cell.BackColor = System.Drawing.Color.Yellow;
                Label lbl = new Label();
                lbl.Text = "<br>Teachers Day!";
                e.Cell.Controls.Add(lbl);

                Image g1 = new Image();
                g1.ImageUrl = "tg.jpg";   // place an image in project folder
                g1.Height = 20;
                g1.Width = 20;
                e.Cell.Controls.Add(g1);
            }
            if (e.Day.Date == new DateTime(2025, 10, 23))
            {
                Calendar1.SelectedDate = new DateTime(2025, 10, 23);
                Calendar1.SelectedDates.SelectRange(Calendar1.SelectedDate,
                    Calendar1.SelectedDate.AddDays(5));

                Label lbl1 = new Label();
                lbl1.Text = "<br>Diwali";
                e.Cell.Controls.Add(lbl1);
            }
        }
        protected void Calendar1_SelectionChanged(object sender, EventArgs e)
        {
            Label1.Text = "Your selected Date: " + Calendar1.SelectedDate.ToShortDateString();
        }
        protected void ButtonReset_Click(object sender, EventArgs e)
        {
            Calendar1.SelectedDate = DateTime.Today;
            Label1.Text = "";
            Label2.Text = "";
            Label3.Text = "";
            Label4.Text = "";
            Label5.Text = "";
        }
    }
}
